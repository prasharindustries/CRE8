Smart Contract Preview



contract CRE8Distribution {

    function distribute(
        address creator,
        address collaborator
    ) external payable {

        uint256 creatorShare =
            (msg.value * 80) / 100;

        uint256 collaboratorShare =
            (msg.value * 20) / 100;

        payable(creator)
            .transfer(creatorShare);

        payable(collaborator)
            .transfer(collaboratorShare);
    }
}
